Questão 10 – IMC
-----------------------------------------------------------------------------------------------------------
#include <stdio.h>

int main()
{
    int Num,NumPlus,NumMore;
    float peso,altura,Calc;
    
    printf("Qual é o peso da pessoa em Kg: ");
    scanf("%f", &peso);
    
    printf("Qual é a altura em metros: ");
    scanf("%f", &altura);
    
    Calc = peso / (altura * altura);
    
    printf("\nÍndice de Massa Corporal [IMC] = %.2f", Calc);
    
    return 0;
}
