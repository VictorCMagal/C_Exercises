#include <stdio.h>

int main()
{
    float V_Total,V_brancos,V_nulo,V_validos,Fracao;
    
    V_Total = 177;
    V_brancos = 18;
    V_nulo = 11;
    V_validos = V_Total - (V_brancos + V_nulo);
    
    printf("Total de Votos = %.f, Votos brancos = %.f e Votos nulos = %.f ",V_Total,V_brancos,V_nulo);

    
    Fracao = (100 * V_validos)/V_Total;
    printf("\nPorsentagem de votos validos = %.2f",Fracao);
    
    Fracao = (100 * V_brancos)/V_Total;
    printf("\nPorsentagem de votos brancos = %.2f",Fracao);
    
    Fracao = (100 * V_nulo)/V_Total;
    printf("\nPorsentagem de votos nulos = %.2f",Fracao);

    return 0;
}
